  <!-- Script for Updating Date and Time -->
  <script>
    let currentLanguage = 'en'; // Default language is English

    function changeLanguage(language) {
      currentLanguage = language;
      updateDateTime();
      // Change the language dropdown to the selected language
      document.getElementById("languageDropdown").innerText = language === 'en' ? 'ENG' : 'NEP';
    }

    function updateDateTime() {
      const now = new Date();

      // English Date Format
      const enDateOptions = { month: 'long', day: '2-digit', year: 'numeric' };
      const enTimeOptions = { hour: '2-digit', minute: '2-digit', hour12: true };
      const enDayOptions = { weekday: 'long' };

      // Nepali Date Format (manually set for Nepali date and time)
      const nepaliDays = ['आइतवार', 'सोमवार', 'मंगलवार', 'बुधबार', 'बिहीवार', 'शुक्रवार', 'शनिवार'];
      const nepaliMonths = ['जनवरी', 'फेब्रुअरी', 'मार्च', 'एप्रिल', 'मे', 'जून', 'जुलाई', 'अगस्ट', 'सेप्टेम्बर', 'अक्टोबर', 'नवम्बर', 'दिसम्बर'];

      // Get English Date/Time
      const enDate = now.toLocaleDateString('en-US', enDateOptions);
      const enTime = now.toLocaleTimeString('en-US', enTimeOptions);
      const enDay = now.toLocaleDateString('en-US', enDayOptions);

      // Get Nepali Date/Time
      const nepaliDate = `${now.getDate()} ${nepaliMonths[now.getMonth()]} ${now.getFullYear()}`;
      const nepaliTime = now.toLocaleTimeString('ne-NP', enTimeOptions);
      const nepaliDay = nepaliDays[now.getDay()];

      // Update date and time based on selected language
      if (currentLanguage === 'en') {
        document.getElementById('date').innerText = enDate;
        document.getElementById('time-day').innerText = `${enTime} / ${enDay}`;
      } else if (currentLanguage === 'ne') {
        document.getElementById('date').innerText = nepaliDate;
        document.getElementById('time-day').innerText = `${nepaliTime} / ${nepaliDay}`;
      }
    }

    setInterval(updateDateTime, 1000); // Update every second
    updateDateTime(); // Initial call to display date and time
  </script>

    function updateCountdown() {
      const endDate = new Date('2024-12-07T00:00:00'); // Set your target date
      const now = new Date();
      const timeLeft = endDate - now;

      if (timeLeft <= 0) {
        document.getElementById('countdown').innerHTML = "Time's up!";
        return;
      }

      const days = Math.floor(timeLeft / (1000 * 60 * 60 * 24));
      const hours = Math.floor((timeLeft % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
      const minutes = Math.floor((timeLeft % (1000 * 60 * 60)) / (1000 * 60));
      const seconds = Math.floor((timeLeft % (1000 * 60)) / 1000);

      document.getElementById('countdown').innerHTML = `${days}d ${hours}h ${minutes}m ${seconds}s`;
